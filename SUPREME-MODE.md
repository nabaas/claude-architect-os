function allocateAgents(task: Task): AgentChain {
  // Step 1: exact trigger match
  const exactMatch = TRIGGER_DICTIONARY.exact(task.firstWord);
  if (exactMatch) return exactMatch.agentChain;

  // Step 2: domain + complexity routing
  const domain = classifyDomain(task);  // uses compound-memory patterns
  const complexity = classifyComplexity(task);

  // Step 3: top-skill agent selection per component
  const agents = AGENT_REGISTRY
    .filter(a => a.domains.includes(domain))
    .sort((a, b) => b.skillScore[domain] - a.skillScore[domain])
    .slice(0, getChainLength(complexity));  // 1-5 agents based on complexity

  // Step 4: add quality gate agents (always)
  if (complexity !== 'trivial') {
    agents.push(AGENTS['code-reviewer']);    // always review
    agents.push(AGENTS['security-engineer']); // always secure
  }

  // Step 5: add self-review at end (always)
  agents.push(AGENTS['self-review']);

  return buildChain(agents, domain);
}
